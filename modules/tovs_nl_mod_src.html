 209  subroutine validateRttovVariable(value, variableName, obsSpaceData, headerIndex, minimum_opt, maximum_opt) 
 210    !
 211    !:Purpose:  check variable for validity for RTTOV-13, 
 212    !            if invalid replace by acceptable value and reject
 213    !
 214    implicit none
 215
 216    ! Arguments:
 217    real(8), intent(inout)          :: value           ! Value of the RTTOV input variable to check for validity
 218    character(len=*),intent(in)     :: variableName    ! Name of the checked variable for output in listings
 219    type(struct_obs), intent(inout) :: obsSpaceData    ! obsSpaceData object
 220    integer, intent(in)             :: headerIndex     ! Index of the observation in obsSpaceData header table 
 221    real(8), intent(in), optional   :: minimum_opt     ! Minimum acceptable value
 222    real(8), intent(in), optional   :: maximum_opt     ! Maximum acceptable value
 223    ! Locals:
 224    real(8)                         :: minimum, maximum
 225
 226    if (present(minimum_opt)) then
 227      minimum = minimum_opt
 228    else
 229      minimum = - huge(0.d0)
 230    end if
 231    
 232    if (present(maximum_opt)) then
 233      maximum = maximum_opt
 234    else
 235      maximum = huge(0.d0)
 236    end if
 237    
 238    if ( value < minimum .or. value > maximum ) then
 239      write(*,*) 'validateRttovVariable: !!! WARNING !!!'
 240      write(*,*) 'validateRttovVariable: INVALID ' // trim(variableName)
 241      write(*,*) 'validateRttovVariable: headerIndex ', headerIndex, " !"
 242      write(*,*) 'validateRttovVariable: ', value, ' should be between ', minimum, ' and ', maximum
 243      value = max(minimum, min(value, maximum) )
 244      write(*,*) 'validateRttovVariable: replaced with ', value, ' !'
 245      call rejectObs(obsSpaceData, headerIndex)
 246    end if
 247
 248  end subroutine validateRttovVariable

 253  subroutine validateRttovProfile(profile, profileName, minimum, maximum, obsSpaceData, headerIndex) 
 254    !
 255    !:Purpose:  check profile for validity for RTTOV-13, 
 256    !            if invalid replace by acceptable value(s) and reject
 257    !
 258    implicit none
 259
 260    ! Arguments:
 261    real(8), intent(inout)          :: profile(:)    ! Vertical profile of input variables to check for validity
 262    character(len=*),intent(in)     :: profileName   ! Name of the checked profile for output in listings
 263    real(8), intent(in)             :: minimum       ! Minimum acceptable value
 264    real(8), intent(in)             :: maximum       ! Maximum acceptable value
 265    type(struct_obs), intent(inout) :: obsSpaceData  ! obsSpaceData object
 266    integer, intent(in)             :: headerIndex   ! Index of the observation in obsSpaceData header table
 267    ! Locals:
 268    logical                         :: ltest(size(profile))
 269    integer                         :: levelIndex
 270    
 271    ltest(:) = (profile(:) > maximum .or. profile(:) < minimum)
 272    
 273    if ( any(ltest) ) then
 274      write(*,*) 'validateRttovProfile: !!! WARNING !!!'
 275      write(*,*) 'validateRttovProfile: some INVALID ' // trim(profileName)
 276      write(*,*) 'validateRttovProfile: headerIndex ', headerIndex, " !"
 277      do levelIndex = 1, size(profile)
 278        if ( ltest(levelIndex) ) then
 279          write(*,*) 'validateRttovProfile: ', profile(levelIndex), &
 280              'should be between ', minimum, ' and ', maximum
 281          profile(levelIndex) = max(minimum, min(profile(levelIndex), maximum) )
 282          write(*,*) 'validateRttovProfile: replaced with ', profile(levelIndex) 
 283        end if
 284      end do
 285      call rejectObs(obsSpaceData, headerIndex)
 286    end if
 287    
 288  end subroutine validateRttovProfile

 317  subroutine tvs_setupAlloc(obsSpaceData)
 318    !
 319    !:Purpose: Memory allocation for the non linear radiative transfer model variables.
 320    !
 321    implicit none
 322
 323    !Arguments:
 324    type(struct_obs) :: obsSpaceData
 325
 326    ! Locals:
 327    integer :: allocStatus(9)
 328    integer :: satelliteCode, instrumentCode, iplatform, isat, instrum
 329    integer :: tovsIndex, idatyp, sensorIndex
 330    integer :: channelNumber, nosensor, channelIndex
 331    integer :: errorStatus
 332    integer :: headerIndex, bodyIndex, taskIndex
 333    logical :: runObsOperatorWithClw
 334    logical :: runObsOperatorWithHydrometeors
 335    logical, allocatable :: logicalBuffer(:)
 336    character(len=32) :: hydroTableFilename
 337
 338    if (tvs_nsensors == 0) return
 339
 340    !  1. Determine the number of radiances to be assimilated.
 341    !      Construct a list of channels for each sensor.
 342    !      Construct a list of sensor number for each profile
 343
 344    write(*,*) 'tvs_setupAlloc: Starting' 
 345
 346    allocStatus = 0
 347    allocate (tvs_nchan(tvs_nsensors),                         stat= allocStatus(1))
 348    allocate (tvs_ichan(tvs_maxNumberOfChannels,tvs_nsensors), stat= allocStatus(2))
 349    allocate (tvs_lsensor(obs_numheader(obsSpaceData)),        stat= allocStatus(3))
 350    allocate (tvs_headerIndex (obs_numheader(obsSpaceData)),   stat= allocStatus(4))
 351    allocate (tvs_tovsIndex(obs_numheader(obsSpaceData)),      stat= allocStatus(5))
 352    allocate (tvs_isReallyPresent(tvs_nsensors),               stat= allocStatus(6))
 353    allocate (tvs_nchanMpiGlobal(tvs_nsensors),                stat= allocStatus(7))
 354    allocate (tvs_ichanMpiGlobal(tvs_maxNumberOfChannels,tvs_nsensors), stat= allocStatus(8))
 355    allocate (tvs_isReallyPresentMpiGlobal(tvs_nsensors), stat= allocStatus(9))
 356
 357    call utl_checkAllocationStatus(allocStatus, ' tvs_setupAlloc')
 358
 359    tvs_nchan(:) = 0 
 360    tvs_ichan(:,:) = 0
 361    tvs_isReallyPresent(:) = .true.
 362    tvs_lsensor(:) = -1
 363    tvs_headerIndex(:) = -1
 364    tvs_tovsIndex (:) = -1
 365
 366    tvs_nobtov = 0
 367
 368    ! Loop over all header indices of the 'TO' family
 369    ! Set the header list & start at the beginning of the list
 370    call obs_set_current_header_list(obsSpaceData,'TO')
 371    HEADER: do
 372      headerIndex = obs_getHeaderIndex(obsSpaceData)
 373      if (headerIndex < 0) exit HEADER
 374
 375      idatyp = obs_headElem_i(obsSpaceData,OBS_ITY,headerIndex)
 376
 377      if ( .not. tvs_isIdBurpTovs(idatyp) ) then
 378        write(*,*) 'tvs_setupAlloc: warning unknown radiance codtyp present check NAMTOVSINST', idatyp
 379        call rejectObs(obsSpaceData, headerIndex)
 380        cycle HEADER   ! Proceed to the next headerIndex
 381      end if
 382      tvs_nobtov = tvs_nobtov + 1
 383     
 384      !    Construct list of channels for each sensor:
 385      !          map burp satellite info to RTTOV platform and satellite.
 386      satelliteCode = obs_headElem_i(obsSpaceData,OBS_SAT,headerIndex)
 387      call tvs_mapSat(satelliteCode,iplatform,isat)
 388      if (iplatform == -1) then
 389        write(*,*) 'Unknown OBS_SAT !', satelliteCode
 390        call utl_abort('tvs_setupAlloc')
 391      end if
 392      !    map burp instrument info to RTTOV instrument.
 393      instrumentCode = obs_headElem_i(obsSpaceData,OBS_INS,headerIndex)
 394      call tvs_mapInstrum(instrumentCode,instrum)
 395      if (instrum == -1) then
 396        write(*,*) 'Unknown OBS_INS !', instrumentCode
 397        call utl_abort('tvs_setupAlloc')
 398      end if
 399      !    find sensor number for this obs.
 400      nosensor =0
 401      do sensorIndex = 1, tvs_nsensors
 402        if ( iplatform == tvs_platforms  (sensorIndex) .and. &
 403             isat      == tvs_satellites (sensorIndex) .and. &
 404             instrum   == tvs_instruments(sensorIndex)      ) then
 405          nosensor = sensorIndex
 406          exit
 407        end if
 408      end do
 409
 410      if (nosensor > 0) then
 411        tvs_lsensor(tvs_nobtov) = nosensor
 412        tvs_headerIndex (tvs_nobtov) = headerIndex
 413        tvs_tovsIndex (headerIndex) = tvs_nobtov
 414      else
 415        write(*,*) ' tvs_setupAlloc: Warning Invalid Sensor ', iplatform, isat, instrum, ' skipping ...'
 416      end if
 417
 418      ! Loop over all body indices (still in the 'TO' family)
 419      ! Set the body list & start at the beginning of the list
 420      call obs_set_current_body_list(obsSpaceData, headerIndex)
 421      BODY: do 
 422        bodyIndex = obs_getBodyIndex(obsSpaceData)
 423        if (bodyIndex < 0) exit BODY
 424        if (nosensor > 0) then
 425          if ( obs_bodyElem_i(obsSpaceData,OBS_ASS,bodyIndex) == obs_assimilated ) then
 426            call tvs_getChannelNumIndexFromPPP( obsSpaceData, headerIndex, bodyIndex, &
 427                                                channelNumber, channelIndex )
 428            if ( channelIndex == 0 ) then
 429              tvs_nchan(nosensor) = tvs_nchan(nosensor) + 1
 430              tvs_ichan(tvs_nchan(nosensor),nosensor) = channelNumber
 431            end if
 432            
 433            if ( tvs_debug .and. mmpi_myid == 0 .and. &
 434                 trim(tvs_instrumentName(nosensor)) == 'AMSUA' ) then
 435              write(*,*) 'test channelNumber:', headerIndex, bodyIndex, nosensor, &
 436                   tvs_satelliteName(nosensor), channelNumber, channelIndex
 437            end if
 438          end if
 439        else           
 440          ! set to notAssimilated if instrument not in NAMTOV namelist
 441          call obs_bodySet_i(obsSpaceData, OBS_ASS, bodyIndex, obs_notAssimilated)
 442        end if
 443      end do BODY
 444    end do HEADER
 445
 446    if ( .not. tvs_userDefinedDoAzimuthCorrection) then 
 447      ! tvs_doAzimuthCorrection user defined values will be overwriten by the old default values 
 448      do sensorIndex = 1, tvs_nsensors
 449        tvs_doAzimuthCorrection(sensorIndex) = ( tvs_platforms(sensorIndex) /= platform_id_eos .and. &
 450             ( tvs_instruments(sensorIndex) == inst_id_amsua .or. tvs_instruments(sensorIndex) == inst_id_mhs )  )     
 451      end do
 452      if ( mmpi_myId == 0 ) write(*,*) ' tvs_setupAlloc: Warning tvs_doAzimuthCorrection user defined values overwriten by the old default values'
 453    end if
 454
 455    if ( .not. tvs_userDefinedIsAzimuthValid ) then 
 456      ! tvs_isAzimuthValid  user defined values will be overwriten by the current default values 
 457      do sensorIndex = 1, tvs_nsensors
 458        tvs_isAzimuthValid(sensorIndex) = .not. ( tvs_isInstrumGeostationary(tvs_instruments(sensorIndex)) )
 459      end do
 460      if ( mmpi_myId == 0 ) write(*,*) ' tvs_setupAlloc: Warning tvs_isAzimuthValid user defined values overwriten by the current default values'
 461    end if
 462
 463    if ( mmpi_myId == 0 ) then
 464      write(*,*) ' tvs_setupAlloc: platform satellite id tvs_doAzimuthCorrection tvs_isAzimuthValid'
 465      do sensorIndex = 1, tvs_nsensors
 466        write(*,'(18x,a,1x,a,1x,i2,1x,L1,10x,L1)') inst_name(tvs_instruments(sensorIndex)), &
 467             platform_name(tvs_platforms(sensorIndex)), tvs_satellites(sensorIndex), &
 468             tvs_doAzimuthCorrection(sensorIndex), tvs_isAzimuthValid(sensorIndex)
 469      end do
 470    end if
 471    
 472    ! Sort list of channels in ascending order.Also force at least one channel, if none are found.
 473    do sensorIndex = 1, tvs_nsensors
 474      call isort(tvs_ichan(:,sensorIndex),tvs_nchan(sensorIndex))
 475      if ( tvs_nchan(sensorIndex) == 0 ) then
 476        tvs_isReallyPresent ( sensorIndex ) =.false.
 477        tvs_nchan(sensorIndex) = 1
 478        tvs_ichan(1,sensorIndex) = 1
 479      end if
 480    end do
 481
 482    write(*,*) ' tvs_setupAlloc: tvs_nobtov = ', tvs_nobtov
 483
 484    do sensorIndex = 1, tvs_nsensors
 485      call tvs_getCommonChannelSet(tvs_ichan(:,sensorIndex),tvs_nchanMpiGlobal(sensorIndex), tvs_ichanMpiGlobal(:,sensorIndex))
 486      print *, 'sensorIndex,tvs_nchan(sensorIndex),tvs_nchanMpiGlobal(sensorIndex)', sensorIndex, tvs_nchan(sensorIndex),tvs_nchanMpiGlobal(sensorIndex)
 487    end do
 488
 489    if (mmpi_myid ==0) then
 490      allocate(logicalBuffer(mmpi_nprocs))
 491    else
 492      allocate(logicalBuffer(1))
 493    end if
 494    
 495    do sensorIndex = 1, tvs_nsensors
 496      call RPN_COMM_gather( tvs_isReallyPresent ( sensorIndex ) , 1, 'MPI_LOGICAL', logicalBuffer, 1,'MPI_LOGICAL', 0, 'GRID', errorStatus )
 497      if (mmpi_myid ==0) then
 498        tvs_isReallyPresentMpiGlobal ( sensorIndex ) =.false.
 499        do taskIndex=1, mmpi_nprocs
 500          tvs_isReallyPresentMpiGlobal ( sensorIndex ) =  tvs_isReallyPresentMpiGlobal ( sensorIndex ) .or. logicalBuffer(taskIndex)
 501        end do
 502      end if
 503      call rpn_comm_bcast(tvs_isReallyPresentMpiGlobal ( sensorIndex ), 1, 'MPI_LOGICAL', 0, 'GRID', errorStatus )
 504    end do
 505    
 506    deallocate(logicalBuffer)
 507
 508    if ( tvs_debug .and. mmpi_myid == 0 ) then
 509      do sensorIndex = 1, tvs_nsensors
 510        write(*,*) 'sensorIndex, tvs_instrumentName(sensorIndex), tvs_satelliteName(sensorIndex)'
 511        write(*,*) sensorIndex, tvs_instrumentName(sensorIndex), tvs_satelliteName(sensorIndex)
 512        write(*,*) 'tvs_channelOffset(sensorIndex), tvs_nchan(sensorIndex)'
 513        write(*,*) tvs_channelOffset(sensorIndex), tvs_nchan(sensorIndex)
 514        write(*,*) 'tvs_ichan(1:tvs_nchan(sensorIndex),sensorIndex)'
 515        write(*,*) tvs_ichan(1:tvs_nchan(sensorIndex),sensorIndex)
 516        write(*,*) 
 517      end do
 518    end if
 519
 520    !  3. Initialize TOVS radiance transfer model
 521
 522    if ( radiativeTransferCode == 'RTTOV' ) then
 523
 524      write(*,'(//,10x,A)') '-rttov_setup: initializing the TOVS radiative transfer model'
 525
 526      allocate (tvs_coefs(tvs_nsensors)          ,stat= allocStatus(1))
 527      allocate (tvs_listSensors (3,tvs_nsensors) ,stat= allocStatus(2))
 528      allocate (tvs_opts (tvs_nsensors)          ,stat= allocStatus(3))
 529      if (tvs_numMWInstrumUsingHydrometeors  > 0) then
 530        allocate (tvs_opts_scatt (tvs_nsensors) ,stat= allocStatus(4))
 531        allocate (tvs_coef_scatt (tvs_nsensors) ,stat= allocStatus(5))
 532      end if
 533      call utl_checkAllocationStatus(allocStatus(1:5), ' tvs_setupAlloc before rttov initialization')
 534
 535      do sensorIndex=1, tvs_nsensors
 536        tvs_listSensors(1,sensorIndex) = tvs_platforms  (sensorIndex)
 537        tvs_listSensors(2,sensorIndex) = tvs_satellites (sensorIndex)
 538        tvs_listSensors(3,sensorIndex) = tvs_instruments(sensorIndex)
 539
 540        runObsOperatorWithClw = (tvs_numMWInstrumUsingCLW /= 0 .and. &
 541                                 tvs_isInstrumUsingCLW(tvs_instruments(sensorIndex)))
 542        runObsOperatorWithHydrometeors = (tvs_numMWInstrumUsingHydrometeors /= 0 .and. &
 543                                          tvs_isInstrumUsingHydrometeors(tvs_instruments(sensorIndex)))
 544
 545        !< General configuration options
 546        tvs_opts(sensorIndex) % config % apply_reg_limits = .true. ! if true application of profiles limits
 547        tvs_opts(sensorIndex) % config % verbose = .false. ! verbose output
 548        tvs_opts(sensorIndex) % config % do_checkinput = .true. ! to check if input profiles are within absolute and regression limits
 549        tvs_opts(sensorIndex) % config % fix_hgpl = .false. ! for backward compatibility with RTTOV-12 should be changed later
 550        !< General RT options
 551        tvs_opts(sensorIndex) % rt_all % switchrad = .true.  ! to use brightness temperature (true) or radiance (false) units in AD routine
 552        tvs_opts(sensorIndex) % rt_all % use_q2m = .false.   ! if true use of surface humidity (false for compatibility with the way rttov 8.7 was compiled)
 553        tvs_opts(sensorIndex) % rt_all % addrefrac = .true.  ! to account for atmospheric refraction
 554        tvs_opts(sensorIndex) % rt_all % dtau_test = .true.  ! for backward compatibility with RTTOV-12 may be changed later
 555        tvs_opts(sensorIndex) % rt_all % use_t2m_opdep = .false. ! for backward compatibility with RTTOV-12 may be changed later
 556        !< VIS/IR RT options
 557        tvs_opts(sensorIndex) % rt_ir % addsolar = .false.  ! to model solar component in the near IR (2000 cm-1 et plus)
 558        tvs_opts(sensorIndex) % rt_ir % addaerosl = .false. ! to account for scattering due to aerosols
 559        tvs_opts(sensorIndex) % rt_ir % addclouds = .false. ! to account for scattering due to clouds
 560        tvs_opts(sensorIndex) % rt_ir % ir_sea_emis_model = 2 ! ISEM (ir_sea_emis_model 1) useful for GEORAD
 561                                                              ! 2 selects IREMIS which is more sophisticated 
 562        tvs_opts(sensorIndex) % rt_ir % pc % ipcreg = -1         ! index of the required PC predictors... to see later
 563        tvs_opts(sensorIndex) % rt_ir % pc % addpc = .false.     ! to carry out principal component calculations 
 564        tvs_opts(sensorIndex) % rt_ir % pc % addradrec = .false. ! to reconstruct radiances from principal components
 565        !< MW RT options
 566        tvs_opts(sensorIndex) % rt_mw % clw_data = tvs_isInstrumUsingCLW(tvs_instruments(sensorIndex)) ! disponibilite du profil d'eau liquide
 567        tvs_opts(sensorIndex) % rt_mw % fastem_version = 6  ! use fastem version 6 microwave sea surface emissivity model (1-6)
 568        tvs_opts(sensorIndex) % rt_mw % clw_scheme = 1 ! default and recommended is 2 just for backward compatibility
 569        !< Interpolation options
 570        tvs_opts(sensorIndex) % interpolation % addinterp = .true. ! use of internal profile interpolator (rt calculation on model levels)
 571        tvs_opts(sensorIndex) % interpolation % lgradp = .true.    ! allow tl/ad of user pressure levels
 572        tvs_opts(sensorIndex) % interpolation % interp_mode = interp_rochon_loglinear_wfn ! see table 9 page 37 of RTTOV 12.1 users guide
 573        tvs_opts(sensorIndex) % interpolation % reg_limit_extrap = tvs_regLimitExtrap 
 574
 575        tvs_opts(sensorIndex) % rt_all % co2_data = .false.
 576        tvs_opts(sensorIndex) % rt_all % n2o_data = .false.
 577        tvs_opts(sensorIndex) % rt_all % co_data  = .false.
 578        tvs_opts(sensorIndex) % rt_all % ch4_data = .false.
 579
 580        if (runObsOperatorWithHydrometeors) then
 581          tvs_opts_scatt(sensorIndex) % interp_mode =  tvs_opts(sensorIndex) % interpolation % interp_mode ! Set interpolation method
 582          tvs_opts_scatt(sensorIndex) % reg_limit_extrap = tvs_regLimitExtrap 
 583          tvs_opts_scatt(sensorIndex) % fastem_version = tvs_opts(sensorIndex) % rt_mw % fastem_version  
 584          tvs_opts_scatt(sensorIndex) % supply_foam_fraction = .false.
 585          tvs_opts_scatt(sensorIndex) % use_t2m_opdep = tvs_opts(sensorIndex) % rt_all % use_t2m_opdep
 586          tvs_opts_scatt(sensorIndex) % use_q2m = tvs_opts(sensorIndex) % rt_all % use_q2m
 587          tvs_opts_scatt(sensorIndex) % lgradp = .true.
 588          tvs_opts_scatt(sensorIndex) % lusercfrac = .false. !< Switch to enable user-specified effective cloud fraction ??
 589          tvs_opts_scatt(sensorIndex) % config % do_checkinput = tvs_opts(sensorIndex) % config % do_checkinput
 590          tvs_opts_scatt(sensorIndex) % config % apply_reg_limits = tvs_opts(sensorIndex) % config % apply_reg_limits
 591          tvs_opts_scatt(sensorIndex) % config % verbose = .true.
 592          tvs_opts_scatt(sensorIndex) % config % fix_hgpl= tvs_opts(sensorIndex) % config % fix_hgpl
 593          ! other option may be considered:
 594          !real(jprb)    :: cc_threshold          = 1.E-3_jprb    !< Minimum effective cloud fraction threshold to consider scattering
 595          !real(jprb)    :: ice_polarisation      = 1.40_jprb     !< Polarised scattering factor for ice hydrometeors (<0 = no polarisation)
 596          !logical(jplm) :: ozone_data            = .false.       !< Switch to enable input of O3 profile
 597                                                                  ! because standard RTTOV coefficients in the MW have no ozone sensitivity
 598          !logical(jplm) :: rad_down_lin_tau      = .true.        !< Linear-in-tau or layer-mean for downwelling radiances
 599          !logical(jplm) :: hydro_cfrac_tlad      = .true.        !< Switch for hydrometeor TL/AD sensitivity to effective cfrac
 600          !logical(jplm) :: zero_hydro_tlad       = .false.       !< Switch for hydrometeor TL/AD sensitivity in layers with zero
 601                                                                  !   hydrometeor concentration
 602        end if
 603        
 604
 605        errorStatus = errorStatus_success
 606        call utl_tmg_start(16,'----RttovSetup')
 607        write(*,*) ' sensorIndex,tvs_nchan(sensorIndex)',  sensorIndex,tvs_nchan(sensorIndex)
 608        if ( tvs_mpiTask0ReadCoeffs ) then
 609          call tvs_rttov_read_coefs(errorStatus, tvs_coefs(sensorIndex), tvs_opts(sensorIndex), & 
 610               tvs_ichan(1:tvs_nchan(sensorIndex),sensorIndex), tvs_listSensors(:,sensorIndex))
 611        else
 612          call rttov_read_coefs (                               &
 613               errorStatus,                                     &! out
 614               tvs_coefs(sensorIndex),                          &
 615               tvs_opts(sensorIndex),                           &
 616               instrument= tvs_listSensors(:,sensorIndex),      &! in
 617               channels=  tvs_ichan(1:tvs_nchan(sensorIndex),sensorIndex) )     ! in option
 618        end if
 619        if (errorStatus /= errorStatus_success) then
 620          write(*,*) 'rttov_read_coefs: fatal error reading coefficients',errorStatus,sensorIndex,tvs_listSensors(1:3,sensorIndex)
 621          call utl_abort('tvs_setupAlloc')
 622        end if
 623       
 624        if (runObsOperatorWithHydrometeors) then
 625          hydrotableFilename = 'hydrotable_' // trim(platform_name(tvs_platforms(sensorIndex))) // '_' // &
 626               trim(inst_name(tvs_instruments(sensorIndex))) // '.dat'
 627          call rttov_read_scattcoeffs(errorStatus, tvs_opts_scatt(sensorIndex), tvs_coefs(sensorIndex), &
 628               tvs_coef_scatt(sensorIndex), file_coef=hydrotableFilename)
 629          if (errorStatus /= errorStatus_success) then
 630            write(*,*) 'rttov_read_scattcoeffs: fatal error reading RTTOV-SCATT coefficients', hydrotableFilename
 631            call utl_abort('tvs_setupAlloc')
 632          end if
 633        end if
 634        call utl_tmg_stop(16)
 635
 636        tvs_opts(sensorIndex) % rt_all % ozone_data = ( tvs_coefs(sensorIndex) % coef % nozone > 0 ) ! profil d'ozone disponible
 637
 638        ! Ensure the options and coefficients are consistent
 639        call rttov_user_options_checkinput(errorStatus, tvs_opts(sensorIndex), tvs_coefs(sensorIndex))
 640        if (errorStatus /= errorStatus_success) then
 641          write(*,*) 'error in rttov options', errorStatus
 642          call utl_abort('tvs_setupAlloc')
 643        end if
 644       
 645      end do
 646
 647
 648      !   4. Memory allocations for radiative tranfer model variables
 649
 650      ! Radiance by profile
 651
 652      allocate( tvs_radiance(tvs_nobtov) ,stat= allocStatus(1))
 653
 654      call utl_checkAllocationStatus(allocStatus(1:1), ' tvs_setupAlloc radiances 1')
 655  
 656      do tovsIndex = 1, tvs_nobtov
 657        sensorIndex = tvs_lsensor(tovsIndex)
 658        if (sensorIndex > -1) then
 659          ! allocate BT equivalent to total direct, tl and ad radiance output
 660          allocate( tvs_radiance(tovsIndex)  % bt  ( tvs_nchan(sensorIndex) ) ,stat= allocStatus(1))
 661          tvs_radiance(tovsIndex)  % bt  ( : ) = 0.d0
 662          call utl_checkAllocationStatus(allocStatus(1:1), ' tvs_setupAlloc radiances 2')
 663          nullify (tvs_radiance(tovsIndex)  % clear )
 664        end if
 665      end do
 666
 667    end if
 668  
 669    write(*,*) 'Leaving tvs_setupAlloc'
 670
 671  end subroutine tvs_setupAlloc

 734  !--------------------------------------------------------------------------
 735  ! tvs_setup
 736  !--------------------------------------------------------------------------
 737  subroutine tvs_setup
 738    !
 739    !:Purpose: to read namelist NAMTOV, initialize the observation error covariance and setup RTTOV-12.
 740    !
 741
 742    implicit none
 743    ! Locals:
 744    integer  sensorIndex, ierr, nulnam
 745    integer, external :: fclos, fnom
 746    integer :: instrumentIndex, numMWInstrumToUseCLW, numMWInstrumToUseHydrometeors
 747
 748    ! Namelist variables: (local)
 749    character(len=8)  :: crtmodl ! For now, must equal 'RTTOV'
 750    integer :: nsensors ! MUST NOT BE INCLUDED IN NAMELIST!
 751    character(len=15) :: csatid(tvs_maxNumberOfSensors)        ! List of satellite names
 752    character(len=15) :: cinstrumentid(tvs_maxNumberOfSensors) ! List of incstrument names
 753    logical :: ldbgtov  ! Choose to print simulated and observed Tb to listing
 754    logical :: useO3Climatology ! Choose to use ozone climatology (otherwise model field)
 755    logical :: regLimitExtrap ! Choose to use RTTOV reg_limit_extrap option
 756    logical :: doAzimuthCorrection(tvs_maxNumberOfSensors) ! Choose to apply correction to azimuth angle
 757    logical :: isAzimuthValid(tvs_maxNumberOfSensors) ! Indicate if azimuth angle is valid
 758    logical :: userDefinedDoAzimuthCorrection ! Indicate if user defined azimuth correction is to be used
 759    logical :: userDefinedIsAzimuthValid ! Indicate if user defined azimuth angle is valid
 760    logical :: mpiTask0ReadCoeffs ! Choose to read coeffs only on task 0 and broadcast
 761    logical :: mwInstrumUsingCLW_tl ! Choose to use CLW increment in TL/AD if exists as state variable
 762    logical :: mwInstrumUsingHydrometeors_tl ! Choose to all hydomet variables in TL/AD if exist as state variables
 763    real(8) :: cloudScaleFactor  ! Scale factor applied to model produced clouds to account for bias
 764    character(len=15) :: instrumentNamesUsingCLW(tvs_maxNumberOfSensors) ! List of inst names using CLW
 765    character(len=15) :: instrumentNamesUsingHydrometeors(tvs_maxNumberOfSensors) ! List of inst name using full set of hydromet variables
 766    logical :: mwAllskyAssim ! High-level key to activate all-sky treatment of MW radiances
 767
 768    namelist /NAMTOV/ nsensors, csatid, cinstrumentid
 769    namelist /NAMTOV/ ldbgtov,useO3Climatology
 770    namelist /NAMTOV/ useUofWIREmiss, crtmodl
 771    namelist /NAMTOV/ useMWEmissivityAtlas, mWAtlasId
 772    namelist /NAMTOV/ mwInstrumUsingCLW_tl, instrumentNamesUsingCLW
 773    namelist /NAMTOV/ mwInstrumUsingHydrometeors_tl, instrumentNamesUsingHydrometeors
 774    namelist /NAMTOV/ regLimitExtrap, doAzimuthCorrection, userDefinedDoAzimuthCorrection
 775    namelist /NAMTOV/ isAzimuthValid, userDefinedIsAzimuthValid, cloudScaleFactor 
 776    namelist /NAMTOV/ mwAllskyAssim, mpiTask0ReadCoeffs
 777
 778    ! return if the NAMTOV does not exist
 779    if ( .not. utl_isNamelistPresent('NAMTOV','./flnml') ) then
 780      write(*,*)
 781      write(*,*) 'tvs_setup: Namelist block NAMTOV is missing in the namelist.'
 782      write(*,*) '           Skipping tvs_setup.'
 783      return
 784    end if
 785 
 786    !   1.1 Default values for namelist variables
 787
 788    nsensors = MPC_missingValue_INT
 789    csatid(:) = '***UNDEFINED***'
 790    cinstrumentid(:) = '***UNDEFINED***'
 791    doAzimuthCorrection(:) = .false.
 792    isAzimuthValid(:) = .false.
 793    ldbgtov = .false.
 794    useO3Climatology = .true.
 795    userDefinedDoAzimuthCorrection = .false.
 796    userDefinedIsAzimuthValid = .false.
 797    crtmodl = 'RTTOV'
 798    useUofWIREmiss = .false.
 799    useMWEmissivityAtlas = .false.
 800    mWAtlasId = 1 !Default to TELSEM-2
 801    mwInstrumUsingCLW_tl = .false.
 802    mwInstrumUsingHydrometeors_tl = .false.
 803    instrumentNamesUsingCLW(:) = '***UNDEFINED***'
 804    instrumentNamesUsingHydrometeors(:) = '***UNDEFINED***'
 805    regLimitExtrap = .true.
 806    cloudScaleFactor = 0.5D0
 807    mwAllskyAssim = .false.
 808    mpiTask0ReadCoeffs = .true.
 809
 810    !   1.2 Read the NAMELIST NAMTOV to modify them
 811 
 812    nulnam = 0
 813    ierr = fnom(nulnam,'./flnml','FTN+SEQ+R/O',0)
 814    read(nulnam, nml=namtov, iostat=ierr)
 815    if (ierr /= 0) call utl_abort('tvs_setup: Error reading namelist NAMTOV')
 816
 817    if (mmpi_myid == 0) write(*,nml=namtov)
 818    ierr = fclos(nulnam)
 819
 820    !  1.3 Transfer namelist variables to module variables
 821    if (nsensors /= MPC_missingValue_INT) then
 822      call utl_abort('tvs_setup: check namelist section NAMTOV; nsensors should be removed as it is' // &
 823          ' now computed by Midas from cinstrumentid and csatid arrays')
 824    end if
 825    
 826    tvs_nsensors = 0
 827    sensor_loop:do sensorIndex = 1, tvs_maxNumberOfSensors
 828      if (cinstrumentid(sensorIndex) /= "***UNDEFINED***" .and. &
 829          csatid(sensorIndex) /= "***UNDEFINED***" ) then
 830        tvs_nsensors = tvs_nsensors + 1
 831      else
 832        exit sensor_loop
 833      end if
 834    end do sensor_loop
 835    
 836    tvs_debug = ldbgtov
 837    radiativeTransferCode = crtmodl
 838    tvs_useO3Climatology = useO3Climatology
 839    tvs_instrumentName(:) = cinstrumentid(:)
 840    tvs_satelliteName(:) = csatid(:)
 841    tvs_mwInstrumUsingCLW_tl = mwInstrumUsingCLW_tl
 842    tvs_regLimitExtrap = regLimitExtrap
 843    tvs_userDefinedDoAzimuthCorrection = userDefinedDoAzimuthCorrection
 844    tvs_userDefinedIsAzimuthValid = userDefinedIsAzimuthValid
 845    tvs_doAzimuthCorrection(:) =  doAzimuthCorrection(:)
 846    tvs_isAzimuthValid(:) =  isAzimuthValid(:)
 847    tvs_cloudScaleFactor = cloudScaleFactor 
 848    tvs_mwAllskyAssim = mwAllskyAssim
 849    tvs_mpiTask0ReadCoeffs = mpiTask0ReadCoeffs
 850
 851    !  1.4 Validate namelist values
 852    
 853    if ( tvs_nsensors == 0 ) then
 854      if(mmpi_myid==0) then 
 855        write(*,*) ' ====================================================='
 856        write(*,*) ' tvs_setup: Number of sensors is zero, skipping setup'
 857        write(*,*) ' ====================================================='
 858      end if
 859      return
 860    end if
 861
 862    if ( radiativeTransferCode /= 'RTTOV' ) then
 863      write(*,'(A)') ' Invalid radiation model name'
 864      call utl_abort('tvs_setup')
 865    end if
 866
 867    if ( tvs_nsensors > tvs_maxNumberOfSensors ) then
 868      write(*,'(A)') ' Number of sensors (tvs_nsensors) is greater than maximum allowed (tvs_maxNumberOfSensors)'
 869      call utl_abort('tvs_setup')
 870    end if
 871
 872    !  1.5 Print the content of this NAMELIST
 873
 874    if(mmpi_myid == 0) then
 875      write(*,'(A)') 
 876      write(*,'(3X,A)') '- Parameters used for TOVS processing (read in NAMTOV)'
 877      write(*,'(3X,A)') '  ----------------------------------------------------'
 878      write(*,'(6X,A,2X,L1)') 'TOVS debug                           : ', tvs_debug
 879      write(*,'(6X,A,2X,L1)') 'Use of UW IR land emissivity atlases : ', useUofWIREmiss
 880      write(*,'(6X,A,2X,L1)') 'Use of MW land emissivity atlases    : ', useMWEmissivityAtlas
 881      if (useMWEmissivityAtlas) then
 882        write(*,'(6X,A,2X,I1)') 'MW atlas Id                          : ', mWAtlasId
 883      end if
 884      write(*,'(6X,A,2X,L1)') 'Use of reg_limit_extrap              : ', regLimitExtrap
 885      write(*,'(6X,A,2X,A)')  'Radiative transfer model             : ', radiativeTransferCode
 886      write(*,'(6X,A,2X,I3)') 'Number of sensors                    : ', tvs_nsensors
 887      write(*,"(6X,'Satellite ids          : ',10A10)") (tvs_satelliteName(sensorIndex), sensorIndex=1,tvs_nsensors)
 888      write(*,"(6X,'Instrument ids         : ',10A10)") (tvs_instrumentName(sensorIndex), sensorIndex=1,tvs_nsensors)
 889      write(*,'(A)') 
 890      write(*,'(A)') 
 891      write(*,'(A)') 
 892      write(*,'(3X,A)') '- Reading and initialization in preparation to the TOVS processing'
 893      write(*,'(5X,A)') '----------------------------------------------------------------'
 894    end if
 895
 896    !  1.6 Set up platform, satellite, instrument and channel mapping
 897
 898    call sensors()
 899
 900    ! Get the name and number of instruments to use CLW
 901    instrumentIdsUsingCLW(:) = -1
 902    do instrumentIndex = 1, tvs_nsensors
 903      instrumentIdsUsingCLW(instrumentIndex) = tvs_getInstrumentId(instrumentNamesUsingCLW(instrumentIndex))
 904      if ( instrumentNamesUsingCLW(instrumentIndex) /= '***UNDEFINED***' ) then
 905        if ( instrumentIdsUsingCLW(instrumentIndex) == -1 ) then
 906          write(*,*) instrumentIndex, instrumentNamesUsingCLW(instrumentIndex)
 907          call utl_abort('tvs_setup: Unknown instrument name to use CLW')
 908        end if
 909      else

2528        !1d-5 was chosen as a threshold because it is the high precision for latitudes in BUFR/BURP files
2529        if (latitudes(profileCount) > 90.d0 .and. (latitudes(profileCount)-90.d0) < 1.d-5) latitudes(profileCount) = 90.d0
2530        if (latitudes(profileCount) < -90.d0 .and. (-latitudes(profileCount)-90.d0) < 1.d-5) latitudes(profileCount) = -90.d0
2531        call validateRttovVariable(latitudes(profileCount), 'latitude', obsSpaceData, headerIndex, -90.d0, 90.d0) 
2532        
2533        profiles(tovsIndex) % longitude = obs_headElem_r(obsSpaceData,OBS_LON,headerIndex) * MPC_DEGREES_PER_RADIAN_R8
2534
2535        surfTypeIsWater(profileCount) = ( tvs_ChangedStypValue(obsSpaceData,headerIndex) == surftype_sea )
2536
2537        do levelIndex = 1, nlv_T
2538          pressure(levelIndex,profileCount) = col_getPressure(columnTrl,levelIndex,headerIndex,'TH') * MPC_MBAR_PER_PA_R8
2539          if ((runObsOperatorWithClw .and. surfTypeIsWater(profileCount)) .or. &
2540              (runObsOperatorWithHydrometeors .and. surfTypeIsWater(profileCount))) then
2541
2542            ! cloud liquid water content
2543            clw(levelIndex,profileCount) = col_getElem(columnTrl,levelIndex,headerIndex,'LWCR')
2544            if (clw(levelIndex,profileCount) < qlim_getMinValueCloud('LWCR') .or. &
2545                clw(levelIndex,profileCount) > qlim_getMaxValueCloud('LWCR')) then
2546              write(*,*) 'tvs_fillProfiles: clw=' , clw(:,profileCount) 
2547              call utl_abort('tvs_fillProfiles: columnTrl has clw outside RTTOV bounds')
2548            end if
2549
2550            clw(levelIndex,profileCount) = clw(levelIndex,profileCount) * tvs_cloudScaleFactor
2551          end if
2552          if (runObsOperatorWithHydrometeors .and. surfTypeIsWater(profileCount)) then
2553            ! cloud ice water content
2554            ciw(levelIndex,profileCount) = col_getElem(columnTrl,levelIndex,headerIndex,'IWCR')
2555            if (ciw(levelIndex,profileCount) < qlim_getMinValueCloud('IWCR') .or. &
2556                ciw(levelIndex,profileCount) > qlim_getMaxValueCloud('IWCR')) then
2557              write(*,*) 'tvs_fillProfiles: ciw=' , ciw(:,profileCount) 
2558              call utl_abort('tvs_fillProfiles: columnTrl has ciw outside RTTOV bounds')
2559            end if
2560
2561            ! rain flux (zero, if not part of control variables)
2562            if (col_varExist(columnTrl,'RF')) then
2563              rainFlux(levelIndex,profileCount) = col_getElem(columnTrl,levelIndex,headerIndex,'RF')
2564            end if
2565            if (rainFlux(levelIndex,profileCount) < qlim_getMinValueCloud('RF') .or. &
2566                rainFlux(levelIndex,profileCount) > qlim_getMaxValueCloud('RF')) then
2567              write(*,*) 'tvs_fillProfiles: rainFlux=' , rainFlux(:,profileCount) 
2568              call utl_abort('tvs_fillProfiles: columnTrl has rain flux outside RTTOV bounds')
2569            end if
2570
2571            ! snow flux (zero, if not part of control variables)
2572            if (col_varExist(columnTrl,'SF')) then
2573              snowFlux(levelIndex,profileCount) = col_getElem(columnTrl,levelIndex,headerIndex,'SF')
2574            end if
2575            if (snowFlux(levelIndex,profileCount) < qlim_getMinValueCloud('SF') .or. &
2576                snowFlux(levelIndex,profileCount) > qlim_getMaxValueCloud('SF')) then
2577              write(*,*) 'tvs_fillProfiles: snowFlux=' , snowFlux(:,profileCount) 
2578              call utl_abort('tvs_fillProfiles: columnTrl has snow flux outside RTTOV bounds')
2579            end if
2580
2581            ! cloud fraction (zero, if not part of control variables)
2582            if (col_varExist(columnTrl,'CLDR')) then
2583              cloudFraction(levelIndex,profileCount) = col_getElem(columnTrl,levelIndex,headerIndex,'CLDR')
2584            end if
2585            if (cloudFraction(levelIndex,profileCount) < qlim_getMinValueCloud('CLDR') .or. &
2586                cloudFraction(levelIndex,profileCount) > qlim_getMaxValueCloud('CLDR')) then
2587              write(*,*) 'tvs_fillProfiles: cloudFraction=' , cloudFraction(:,profileCount) 
2588              call utl_abort('tvs_fillProfiles: columnTrl has cloud fraction outside RTTOV bounds')
2589            end if
2590
2591            ciw(levelIndex,profileCount) = ciw(levelIndex,profileCount) * tvs_cloudScaleFactor
2592            rainFlux(levelIndex,profileCount) = rainFlux(levelIndex,profileCount) * tvs_cloudScaleFactor
2593            snowFlux(levelIndex,profileCount) = snowFlux(levelIndex,profileCount) * tvs_cloudScaleFactor
2594
2595          end if ! runObsOperatorWithHydrometeors .and. surfTypeIsWater
2596        end do ! levelIndex
2597        
2598        if (tvs_coefs(sensorIndex) %coef % nozone > 0 .and. .not. tvs_useO3Climatology) then
2599          column_ptr => col_getColumn(columnTrl, headerIndex, trim(ozoneVarName) )
2600          ! Conversion from microgram/km to ppmv (to have the same units as climatology when tvs_useO3Climatology is .true.
2601          ! Conversion to kg/kg for use by RTTOV in done later
2602          ozone(:,profileCount) = column_ptr(:) * 1.0D-9 * o3Mixratio2ppmv
2603        end if
2604
2605      end do bobs2
2606
2607      !    2.5  Get ozone profiles (ppmv) from climatology if necessary
2608      if (tvs_coefs(sensorIndex) %coef % nozone > 0 .and. tvs_useO3Climatology) then
2609        call ozo_get_profile (ozone, latitudes, pressure, nlv_T, profileCount)
2610      end if
2611
2612      !   2.5  Fill profiles structure
2613
2614      do  profileIndex = 1 , profileCount 
2615        tovsIndex = sensorTovsIndexes(profileIndex)
2616        headerIndex = sensorHeaderIndexes(profileIndex)
2617        profiles(tovsIndex) % gas_units       = gas_unit_specconc ! all gas profiles are supposed to be provided in kg/kg (specific humidity, i.e. mass mixing ratio [kg/kg] over wet air)
2618        profiles(tovsIndex) % id              = '' ! profile id, up to 128 characters, to consider for use
2619        profiles(tovsIndex) % nlevels         = nlv_T
2620        profiles(tovsIndex) % nlayers         = nlv_T - 1
2621        profiles(tovsIndex) % date(1)         = year
2622        profiles(tovsIndex) % date(2)         = month
2623        profiles(tovsIndex) % date(3)         = day
2624        profiles(tovsIndex) % latitude        = latitudes(profileIndex)
2625        profiles(tovsIndex) % elevation       = 0.001d0 * col_getHeight(columnTrl,ilowlvl_T,headerIndex,'TH') ! unite km
2626        call validateRttovVariable(profiles(tovsIndex) % elevation, "elevation", obsSpaceData, headerIndex, maximum_opt=elevmax)
2627        profiles(tovsIndex) % skin % watertype = watertype_ocean_water !utilise pour calcul rayonnement solaire reflechi seulement
2628        profiles(tovsIndex) % skin % t         = col_getElem(columnTrl,1,headerIndex,'TG')
2629        call validateRttovVariable( profiles(tovsIndex) % skin % t, "skin temperature", obsSpaceData, headerIndex, tmin, tmax) 
2630        profiles(tovsIndex) % skin % salinity  = 35.d0 ! for FASTEM-4 only to revise (practical salinity units)
2631        profiles(tovsIndex) % skin % fastem(:) = 0.0d0
2632        profiles(tovsIndex) % skin % snow_fraction  = 0.d0 ! Surface coverage snow fraction(0-1), used only by IR emissivity atlas
2633        profiles(tovsIndex) % skin % soil_moisture  = 0.d0 ! soil moisure (m**3/m**3) not yet used
2634        profiles(tovsIndex) % s2m % t          = col_getElem(columnTrl,ilowlvl_T,headerIndex,'TT')
2635        call validateRttovVariable(profiles(tovsIndex) % s2m % t, '2m air temperature', &
2636            obsSpaceData, headerIndex, tmin, tmax) 
2637        profiles(tovsIndex) % s2m % p         = col_getElem(columnTrl,1      ,headerIndex,'P0')*MPC_MBAR_PER_PA_R8
2638        call validateRttovVariable(profiles(tovsIndex) % s2m % p, 'surface pressure', &
2639            obsSpaceData, headerIndex, pmin, pmax) 
2640        profiles(tovsIndex) % s2m % u         = col_getElem(columnTrl,ilowlvl_M,headerIndex,'UU')
2641        profiles(tovsIndex) % s2m % v         = col_getElem(columnTrl,ilowlvl_M,headerIndex,'VV')
2642        wind = sqrt( profiles(tovsIndex) % s2m % u ** 2 + &
2643            profiles(tovsIndex) % s2m % v ** 2 )
2644        if ( wind > wmax ) then
2645          write(*,*) 'tvs_fillProfiles: !!! WARNING !!!'
2646          write(*,*) 'tvs_fillProfiles: INVALID 10m wind speed'
2647          write(*,*) 'tvs_fillProfiles: headerIndex ', headerIndex, " !"
2648          write(*,*) 'tvs_fillProfiles: modulus ', wind, ' larger than ', wmax, 'set to zero !'
2649          profiles(tovsIndex) % s2m % u = 0.d0
2650          profiles(tovsIndex) % s2m % v = 0.d0
2651          call rejectObs(obsSpaceData, headerIndex)
2652        end if
2653        profiles(tovsIndex) % s2m % o         = 0.0d0 !surface ozone never used
2654        profiles(tovsIndex) % s2m % wfetc     = 100000.0d0 ! Wind fetch (in meter for rttov10 ?) used to calculate reflection of solar radiation by sea surface
2655        profiles(tovsIndex) % icede_param     = 0
2656        profiles(tovsIndex) % Be              = 0.4d0 ! earth magnetic field strength (gauss) (must be non zero)
2657        profiles(tovsIndex) % cosbk           = 0.0d0 ! cosine of the angle between the earth magnetic field and wave propagation direction
2658        profiles(tovsIndex) % p(:)            = pressure(:,profileIndex)
2659        call validateRttovVariable(profiles(tovsIndex) % p(nlv_T), "pressure profile near surface", obsSpaceData, headerIndex, maximum_opt=2000.d0)
2660        call validateRttovVariable(profiles(tovsIndex) % p(1), "pressure profile near top", obsSpaceData, headerIndex, 0.d0) 
2661        !RTTOV scatt needs half pressure levels (see figure 5 of RTTOV 12 User's Guide)
2662        if (runObsOperatorWithHydrometeors) then
2663          cld_profiles(tovsIndex) % ph (1) = 0.d0
2664          cld_profiles(tovsIndex) % cfrac = 0.d0
2665          do levelIndex = 1, nlv_T - 1
2666            cld_profiles(tovsIndex) % ph (levelIndex+1) = 0.5d0 * (profiles(tovsIndex) % p(levelIndex) + &
2667                                                                   profiles(tovsIndex) % p(levelIndex+1))
2668          end do
2669          cld_profiles(tovsIndex) % ph (nlv_T+1) = profiles(tovsIndex) % s2m % p
2670        end if
2671        column_ptr => col_getColumn(columnTrl, headerIndex,'TT' )
2672        profiles(tovsIndex) % t(:)   = column_ptr(:)
2673        call validateRttovProfile(profiles(tovsIndex) % t, 'temparature', tmin, tmax, obsSpaceData, headerIndex) 
2674        if (tvs_coefs(sensorIndex) %coef %nozone > 0) then
2675          profiles(tovsIndex) % o3(:) = ozone(:,profileIndex) * o3ppmv2Mixratio ! Climatology output is ppmv over dry air                                                                                                            ! because atmosphere is very dry where there is significant absorption by ozone)
2676          if (.not. tvs_useO3Climatology)  then
2677            profiles(tovsIndex) % s2m % o  = col_getElem(columnTrl,ilowlvl_T,headerIndex,trim(ozoneVarName)) * 1.0d-9 ! Assumes model ozone in ug/kg
2678          end if
2679          call validateRttovProfile(profiles(tovsIndex) % o3, 'ozone', o3min, o3max, obsSpaceData, headerIndex)
2680        end if
2681
2682        column_ptr => col_getColumn(columnTrl, headerIndex,'HU' )
2683        profiles(tovsIndex) % q(:)            =  column_ptr(:)
2684        call validateRttovProfile(profiles(tovsIndex) % q, 'water vapor', qmin, qmax, obsSpaceData, headerIndex)
2685        profiles(tovsIndex) % ctp = 1013.25d0
2686        profiles(tovsIndex) % cfraction = 0.d0
2687        if (runObsOperatorWithClw) then
2688          profiles(tovsIndex) % clw(:) = clw(:,profileIndex)
2689        else if (runObsOperatorWithHydrometeors) then
2690          cld_profiles(tovsIndex) % hydro(:,1) = rainFlux(:,profileIndex)
2691          cld_profiles(tovsIndex) % hydro(:,2) = snowFlux(:,profileIndex)
2692          cld_profiles(tovsIndex) % hydro(:,4) = clw(:,profileIndex)
2693          cld_profiles(tovsIndex) % hydro(:,5) = ciw(:,profileIndex)
2694
2695          do levelIndex = 1, nlv_T
2696            if (cld_profiles(tovsIndex) % hydro(levelIndex,1) > qlim_getMinValueCloud('RF') .or. &
2697                cld_profiles(tovsIndex) % hydro(levelIndex,2) > qlim_getMinValueCloud('SF') .or. &
2698                cld_profiles(tovsIndex) % hydro(levelIndex,4) > qlim_getMinValueCloud('LWCR') .or. &
2699                cld_profiles(tovsIndex) % hydro(levelIndex,5) > qlim_getMinValueCloud('IWCR')) then
2700              
2701              ! set to overcast cloud, if CLDR not part of control variables
2702              if (col_varExist(columnTrl,'CLDR')) then
2703                cld_profiles(tovsIndex) % hydro_frac(levelIndex,1) = cloudFraction(levelIndex,profileIndex)
2704              else
2705                cld_profiles(tovsIndex) % hydro_frac(levelIndex,1) = 1.0d0
2706              end if
2707            else
2708              cld_profiles(tovsIndex) % hydro_frac(levelIndex,1) = qlim_getMinValueCloud('CLDR')  
2709            end if
2710
2711          end do ! levelIndex
2712        end if  ! runObsOperatorWithClw
2713      end do ! profileIndex
2714
2715      deallocate (pressure,            stat = allocStatus(2))
2716      deallocate (ozone,               stat = allocStatus(3))
2717      deallocate (latitudes,           stat = allocStatus(4))
2718      deallocate (sensorHeaderIndexes, stat = allocStatus(5))
2719      deallocate (sensorTovsIndexes,   stat = allocStatus(6))
2720      if (tvs_coefs(sensorIndex) %coef %nozone > 0 .and. .not.tvs_useO3Climatology) then
2721        deallocate (ozone,             stat = allocStatus(7))
2722      end if
2723      if ( allocated(clw) ) then
2724        deallocate (clw,      stat= allocStatus(8))
2725      end if
2726      if ( allocated(ciw) ) then
2727        deallocate (ciw,      stat= allocStatus(9))
2728        deallocate (rainFlux, stat= allocStatus(10))
2729        deallocate (snowFlux, stat= allocStatus(11))
2730        deallocate (cloudFraction, stat= allocStatus(12))
2731      end if
2732      deallocate (surfTypeIsWater,stat= allocStatus(13))
2733
2734      call utl_checkAllocationStatus(allocStatus, ' tvs_fillProfiles', .false.)
2735     
2736    end do sensor_loop
2737
2738  end subroutine tvs_fillProfiles

2776  !--------------------------------------------------------------------------
2777  !  tvs_rttov
2778  !--------------------------------------------------------------------------
2779  subroutine tvs_rttov(obsSpaceData, bgckMode, beSilent)
2780    !
2781    !:Purpose: Interface for RTTOV non linear operator
2782    !           tvs_fillProfiles should be called before
2783    !
2784    implicit none
2785
2786    ! Arguments
2787    type(struct_obs), intent(inout) :: obsSpaceData  ! obsSpaceData structure
2788    logical, intent(in)             :: bgckMode       ! flag to transfer transmittances and cloudy overcast radiances in bgck mode 
2789    logical, intent(in)             :: beSilent       ! flag to control verbosity
2790
2791    ! Locals:
2792    integer :: nlv_T
2793    integer :: btCount
2794    integer :: allocStatus(4)
2795    integer :: rttov_err_stat ! rttov error return code
2796    integer, external :: omp_get_num_threads
2797    integer :: nthreads,max_nthreads
2798    integer :: sensorId, tovsIndex
2799    integer :: channelIndex, channelIndexFound, channelNumber
2800    integer :: profileCount
2801    integer :: profileIndex, levelIndex, jj, btIndex
2802    integer :: instrum
2803    integer :: sensorType        ! sensor type (1=infrared; 2=microwave; 3=high resolution,4=polarimetric)
2804    integer, allocatable:: sensorTovsIndexes(:)  
2805    
2806    type (rttov_emissivity), pointer :: emissivity_local(:)    ! emissivity structure with input and output
2807    type (rttov_chanprof), pointer :: chanprof(:)
2808    type (rttov_chanprof), allocatable :: chanprof1(:)
2809    type (rttov_radiance) :: radiancedata_d, radiancedata_d1
2810    type (rttov_transmission) :: transmission, transmission1
2811    integer              :: asw
2812    logical, pointer :: calcemis(:)
2813    real(8), allocatable  :: surfem1(:)
2814    integer, allocatable  :: frequencies(:)
2815    real(8), allocatable  :: uOfWLandWSurfaceEmissivity(:)
2816    logical, allocatable  :: lchannel_subset(:,:)
2817    integer               :: profileIndex2, tb1, tb2
2818    integer :: istart, iend, bodyIndex, headerIndex
2819    real(8) :: clearMwRadiance
2820    logical :: ifBodyIndexFound
2821    logical :: runObsOperatorWithClw
2822    logical :: runObsOperatorWithHydrometeors
2823
2824    if ( .not. beSilent ) write(*,*) 'tvs_rttov: Starting'
2825    if ( .not. beSilent ) write(*,*) 'Memory Used: ',get_max_rss()/1024,'Mb'
2826
2827    if (tvs_nobtov == 0) return                  ! exit if there are not tovs data
2828
2829    !   1.  Get number of threads available and allocate memory for some variables
2830    !$omp parallel
2831    max_nthreads = omp_get_num_threads()
2832    !$omp end parallel
2833    allocStatus(:) = 0
2834    allocate(sensorTovsIndexes(tvs_nobtov),stat=allocStatus(1))
2835    call utl_checkAllocationStatus(allocStatus(1:1), ' tvs_rttov sensorTovsIndexes')
2836    
2837    
2838    !   1.1   Read surface information
2839    if ( bgckMode ) call EMIS_READ_CLIMATOLOGY
2840
2841    !   2.  Computation of hx for tovs data only
2842
2843    ! Loop over all sensors specified by user
2844    sensor_loop:do sensorId = 1, tvs_nsensors
2845   
2846      sensorType = tvs_coefs(sensorId) % coef % id_sensor
2847      instrum = tvs_coefs(sensorId) % coef % id_inst
2848
2849      runObsOperatorWithClw = (tvs_numMWInstrumUsingCLW /= 0 .and. &
2850                               tvs_isInstrumUsingCLW(tvs_instruments(sensorId)))
2851      runObsOperatorWithHydrometeors = (tvs_numMWInstrumUsingHydrometeors /= 0 .and. &
2852                                        tvs_isInstrumUsingHydrometeors(tvs_instruments(sensorId)))
2853                                        
2854      if (runObsOperatorWithClw .and. runObsOperatorWithHydrometeors) then
2855        call utl_abort('tvs_rttov: this instrument is mentioned in using CLW and hydrometeors.')
2856      end if
2857    
2858      !  loop over all obs.
2859      profileCount = 0
2860      obs_loop: do tovsIndex = 1, tvs_nobtov
2861        
2862        !    Currently processed sensor?
2863        if ( tvs_lsensor(tovsIndex) == sensorId ) then
2864          profileCount = profileCount + 1
2865          sensorTovsIndexes(profileCount) = tovsIndex
2866          nlv_T = tvs_profiles_nl(tovsIndex) % nlevels
2867        end if
2868      end do obs_loop
2869      
2870      if (profileCount == 0) cycle sensor_loop
2871      
2872      !    2.1  Calculate the actual number of threads which will be used.
2873      
2874      nthreads = min(max_nthreads, profileCount )  
2875      
2876      !    2.2  Prepare all input variables required by rttov.
2877      
2878      if ( bgckMode .and. tvs_isInstrumHyperSpectral(instrum) ) then
2879        btCount = profileCount * tvs_nchan(sensorId)
2880      else
2881        btCount = tvs_countRadiances(sensorTovsIndexes(1:profileCount), obsSpaceData)
2882      end if
2883      
2884      if (allocated(tvs_bodyIndexFromBtIndex)) deallocate(tvs_bodyIndexFromBtIndex)
2885      allocate(tvs_bodyIndexFromBtIndex(btCount))
2886
2887      if ( btCount == 0 ) cycle sensor_loop
2888      tvs_bodyIndexFromBtIndex(:) = -1      
2889      allocate ( surfem1          (btCount) ,stat=allocStatus(1))
2890
2891      asw = 1 ! Allocation
2892      call rttov_alloc_direct(            &
2893              allocStatus(2),             &
2894              asw,                        &
2895              nprofiles=profileCount,     & ! (not used)
2896              nchanprof=btCount,          &
2897              nlevels=nlv_T,              &
2898              chanprof=chanprof,          &
2899              opts=tvs_opts(sensorId),    &
2900              coefs=tvs_coefs(sensorId),  &
2901              transmission=transmission,  &
2902              radiance=radiancedata_d,    &
2903              calcemis=calcemis,          &
2904              emissivity=emissivity_local,&
2905              init=.true.)
2906
2907      if (runObsOperatorWithHydrometeors) then
2908        allocate ( frequencies(btCount), stat=allocStatus(3))
2909      end if
2910
2911      if (useUofWIREmiss) then
2912        allocate ( uOfWLandWSurfaceEmissivity(btCount), stat=allocStatus(4) )
2913      end if
2914      call utl_checkAllocationStatus(allocStatus, ' tvs_rttov')
2915      
2916      !     get Hyperspectral IR emissivities
2917      if ( tvs_isInstrumHyperSpectral(instrum) ) then
2918        surfem1(:) = 0.
2919        if ( bgckMode ) then
2920          call emis_getIrEmissivity (surfem1,tvs_nchan(sensorId),sensorId,profileCount,btCount,sensorTovsIndexes)
2921        else
2922          call tvs_getHIREmissivities(sensorTovsIndexes(1:profileCount), obsSpaceData, surfem1)
2923        end if
2924      end if
2925      
2926      if ( bgckMode .and. tvs_isInstrumHyperSpectral(instrum) ) then
2927        btIndex = 0
2928        do profileIndex = 1 , profileCount
2929          do  channelIndex = 1,tvs_nchan(sensorId)
2930            btIndex = btIndex + 1
2931            chanprof(btIndex)%prof = profileIndex
2932            chanprof(btIndex)%chan = channelIndex
2933          end do
2934        end do
2935        
2936        do profileIndex = 1, profileCount
2937          headerIndex = tvs_headerIndex(sensorTovsIndexes(profileIndex))
2938          if (headerIndex > 0) then
2939            istart = obs_headElem_i(obsSpaceData,OBS_RLN,headerIndex)
2940            iend = obs_headElem_i(obsSpaceData,OBS_NLV,headerIndex) + istart - 1
2941            do bodyIndex = istart, iend

4117  subroutine emi_sea(em_oc, wnum, angle, wind, np, nc)
4118    !
4119    !:Purpose: GET OCEAN SURFACE EMISSIVITY
4120    !:Note:    IMEM(NC), set to zero initially, on next call IMEM will have the
4121    !           right boundary channel to save search time in interpolation.
4122    !           IOPT=1 means activate IMEM option (all calls ask for same channels)
4123    !
4124    !           To get surface ocean emissivity for a group of channels with
4125    !           wavenumbers WNUM (cm-1) looking at one point with surface
4126    !           wind speed wind from angle angle.
4127    !           Based on Masuda,1988, Remote Sens. of Envir, 313-329.
4128    !           Coded emissivity routine based on Masuda's data by Tom Kleespies
4129    !           Covers 650-2857 cm-1 or 3.1-15.4 microns
4130    !
4131    !:CAUTION: extrapolated values from 769-650 cm-1
4132    !           and interpolated values between 2439-1250 cm-1
4133    !
4134    implicit none
4135
4136    ! Arguments:
4137    real(8), intent(out)   :: em_oc(nc,np) ! Ocean emissivities (0.-1.)
4138    real(8), intent(in)    :: wnum(nc)     ! Channel wavenumbers (cm-1)
4139    real(8), intent(in)    :: angle(np)    ! Viewing angle (deg)
4140    real(8), intent(in)    :: wind(np)     ! Surface wind speed (m/s)
4141    integer, intent(in)    :: np           ! Number of profiles
4142    integer, intent(in)    :: nc           ! Number of channels
4143
4144    ! Locals
4145    integer     :: i, k, l
4146    integer     :: imem(nc) 
4147    integer     :: mchan(2)
4148    real(8)     :: dum
4149    real(8)     :: emi2(2,np)
4150
4151    ! Masuda's 19 wavelengths converted to wavenumber
4152    real(8), parameter :: refw(19)=(/ 2857.1d0, 2777.7d0, 2702.7d0, 2631.6d0, 2564.1d0, &
4153         2500.0d0, 2439.0d0, 1250.0d0, 1190.5d0, 1136.3d0,                           &
4154         1087.0d0, 1041.7d0, 1000.0d0, 952.38d0, 909.09d0,                           &
4155         869.57d0, 833.33d0, 800.00d0, 769.23d0/)
4156
4157
4158    ! imem options
4159
4160    imem(:) = 0
4161
4162    do I = 1, nc
4163
4164      !  out of range
4165      if ( wnum(I) < 645.d0 .or. wnum(I) > refw(1) ) then
4166        write(*,'(A,1x,e12.4)') ' fatal: wavenumber out of range in emi_sea', wnum(I)
4167        stop
4168      else if ( wnum(I) <= refw(19) .and. wnum(I) > 645.d0 ) then
4169        !  extrapolated from 769 cm-1 to 645 cm-1: NOT FROM REAL DATA
4170        !  nevertheless thought to be much better than unity
4171        !  this is a region of relatively rapid emissivity change
4172        !  worst estimates for 700-645 cm-1, but these channels do not
4173        !  see the surface (strong co2 absorption).
4174        imem(I) = 18
4175      else
4176        !  CAUTION interpolation on large interval 1250-2439 cm-1
4177        !  where no data is available except that of ASTER. ASTER
4178        !  shows a relatively smooth variation with wavelength except
4179        !  for a sharp drop at 1600 cm-1 with highs at 1550 and 1650 cm-1
4180        !  with peak-to-peak variation of 1.5% in that narrow range.
4181        !  Worst estimates would be between 1400-1800 cm-1 in HIRS ch 12
4182        !  which only in very cold atmospheres sees the surface.
4183        do k = 1, 18
4184          if ( wnum(I) > refw(k + 1) .and. wnum(I) <= refw(k) ) then
4185            imem(I) = k
4186          end if
4187        end do
4188
4189      end if
4190   
4191      mchan(1)= imem(I)
4192      mchan(2)= imem(I) + 1
4193
4194      dum = ( wnum(I) - refw(mchan(1)) ) / ( refw(mchan(2)) - refw(mchan(1)) )
4195
4196      call COMP_IR_EMISS(emi2, wind,angle,2,np,mchan)
4197
4198      ! interpolation/extrapolation in wavenumber 
4199
4200      do L = 1, np
4201  
4202        em_oc(I,L) = emi2(1,L) + ( emi2(2,L) - emi2(1,L) ) * dum
4203          
4204      end do
4205
4206    end do
4207
4208
4209  end subroutine emi_sea
